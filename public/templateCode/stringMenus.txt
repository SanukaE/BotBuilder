import StringMenuType from '#types/StringMenuType.js';

const stringMenu: StringMenuType = {
  customID: 'example-menu',
  isDevOnly: true,
  enableDebug: true,
    
  async script(client, interaction, debugStream) {
    const selectedValue = interaction.values[0];
    debugStream.debug(`Selected value: ${selectedValue}`);

    await interaction.followUp({
        content: `You selected: ${selectedValue}`,
        ephemeral: true
    });
    debugStream.write("Follow up sent!");
  },
};

export default stringMenu;
