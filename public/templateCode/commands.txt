import CommandType from '../../../utils/CommandType.js';

const command: CommandType = {
  name: '{{actionName}}',
  description: 'This is a template code. Save the file to view changes live!',
  isDevOnly: true,
  enableDebug: true,

  async script(client, interaction, debugStream) {

      ephemeral: true

  },
