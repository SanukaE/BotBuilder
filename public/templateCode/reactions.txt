import ReactionType from '#types/ReactionType.js';

const reaction: ReactionType = {
  name: '{{actionName}}',
  isDevOnly: true,
  enableDebug: true,

  async script(client, messageReaction, user, details, debugStream) {
    await messageReaction.message.reply({
      content: "Edit the {{fileName}} and view changes live!",
      allowedMentions: { repliedUser: false }
    });

    debugStream.write("Reply sent!");
  },
};

export default reaction;
